    <script>
        const ctx = document.getElementById('privacyChart').getContext('2d');
        let currentAmount = 50000;
        
        const privacyChart = new Chart(ctx, {
            type: 'bar',
            data: {
                labels: ['Public DEX (Exposed)', 'GoDark (Stealth Pool)'],
                datasets: [{
                    label: 'Slippage + MEV Bot Slippage ($ Saved)',
                    data: [ (currentAmount * 0.007).toFixed(0), currentAmount ],
                    backgroundColor: ['#ef4444', '#00ff66'],
                    borderRadius: 12,
                    borderSkipped: false,
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: { legend: { display: false } },
                scales: {
                    y: { display: false, grid: { display: false } },
                    x: { ticks: { color: '#9ca3af' }, grid: { display: false } }
                }
            }
        });

        document.getElementById('simulatorRange').addEventListener('input', function(e) {
            const val = parseInt(e.target.value);
            document.getElementById('rangeValue').innerText = '$' + val.toLocaleString();
            
            const dynamicLoss = (val * 0.007).toFixed(0);
            document.getElementById('lossValue').innerText = '-$' + parseInt(dynamicLoss).toLocaleString();
            
            // Update chart dynamically
            privacyChart.data.datasets[0].data = [dynamicLoss, val];
            privacyChart.update();
        });
    </script>
